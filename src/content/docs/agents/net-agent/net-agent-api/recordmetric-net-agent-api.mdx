<div>
  <h2>Syntax</h2>
  <pre>
    NewRelic.Api.Agent.NewRelic.RecordMetric(<em>string</em> $metric_name,{' '}
    <em>single</em> $metric_value)
  </pre>
  Records a custom metric with the given name.<h2>Requirements</h2>
  <p>Compatible with all agent versions.</p>
  <p>Compatible with all app types.</p>
  <h2>Description</h2>
  <p>
    Records a{' '}
    <a href="/docs/agents/manage-apm-agents/agent-metrics/custom-metrics">
      custom metric
    </a>{' '}
    with the given name. To view these custom metrics, use the{' '}
    <a href="/docs/query-your-data/explore-query-data/query-builder/use-advanced-nrql-mode-specify-data">
      query builder
    </a>{' '}
    to search metrics and create customizable charts. See also{' '}
    <a href="/docs/agents/net-agent/net-agent-api/incrementcounter-net-agent">
      <code>IncrementCounter()</code>
    </a>{' '}
    and{' '}
    <a href="/docs/agents/net-agent/net-agent-api/recordresponsetimemetric-net-agent">
      <code>RecordResponseTimeMetric()</code>
    </a>
    .
  </p>
  <div className="callout-important">
    <p>
      When creating a custom metric, start the name with <code>Custom/</code>{' '}
      (for example, <code>Custom/MyMetric</code>).
    </p>
  </div>
  <h2>Parameters</h2>
  <table>
    <thead>
      <tr>
        <th width="25%">Parameter</th>
        <th>Description</th>
      </tr>
    </thead>
    <tbody>
      <tr>
        <td>
          <p>
            <code>$metric_name</code>
          </p>
          <p>
            <em>string</em>
          </p>
        </td>
        <td>
          Required. The name of the metric to record. Only the first 255
          characters are retained.
        </td>
      </tr>
      <tr>
        <td>
          <p>
            <code>$metric_value</code>
          </p>
          <p>
            <em>single</em>
          </p>
        </td>
        <td>
          <p>Required. The quantity to record for the metric.</p>
        </td>
      </tr>
    </tbody>
  </table>
  <h2>Examples</h2>
  <h3 id="record-stopwatch">Record response time of a sleeping process</h3>
  <pre>
    Stopwatch stopWatch = Stopwatch.StartNew();{'\n'}
    System.Threading.Thread.Sleep(5000);{'\n'}stopWatch.Stop();{'\n'}
    NewRelic.Api.Agent.NewRelic.RecordMetric("Custom/DEMO_Record_Metric",
    stopWatch.ElapsedMilliseconds);
  </pre>
